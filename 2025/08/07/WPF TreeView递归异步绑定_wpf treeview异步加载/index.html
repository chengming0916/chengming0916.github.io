WPF的TreeView不像WinForms的TreeView那样，通过TreeNode实现树型节点。WPF的TreeView实现起来，我感觉有点像VM(View-Model)模式，就像WinForms中的DataGrid，控件负责显示，数据在DataTable中。那么我们在实现TreeView时，也需要用这样的思路，数据在Model中，控件是View负责显示。

我需要实现的最终目标是：
1.节点是异步加载的（解决效率问题）
2.节点是无限级别的（递归的，这样可复用与目录结构、组织架构等数据类型）
3.节点是可以多选的（用CheckBox选择）
4.XAML方式绑定（减少UI上Code量）
5.方便得获取选中项信息

**4）创建TreeView 的Model类DepartmentViewModel

  1. using System;   

  2. using System.Collections.Generic;   

  3. using System.Linq;   

  4. using System.Text;   

  5. using System.ComponentModel;   

  6. using System.Collections.ObjectModel;   

  7. namespace DepartmentTreeView {   

  8.     public class DepartmentViewModel:INotifyPropertyChanged {   

  9.         static readonly DepartmentViewModel _temp = new DepartmentViewModel(null);   

  10.         private static ObservableCollection _checkedItems = new ObservableCollection();   

  11.         static DepartmentViewModel _rootItem;  

  12.         #region fields&properties   

  13.         private bool? _isChecked;   

  14.         public bool? IsChecked {   

  15.             get { return _isChecked; }   

  16.             set {   

  17.                 SetCheckState(value, true, true);   

  18.             }   

  19.         }   

  20.         private void SetCheckState(bool? value, bool updateChildren, bool updateParent) {   

  21.             if (_isChecked != value) {   

  22.                 _isChecked = value;   

  23.                 if (_isChecked == true) {   

  24.                     _checkedItems.Add(this);   

  25.                     PropertyChanged(this, new PropertyChangedEventArgs(“CheckedItems”));   

  26.                 } else if (_isChecked == false) {   

  27.                     _checkedItems.Remove(this);   

  28.                     PropertyChanged(this, new PropertyChangedEventArgs(“CheckedItems”));   

  29.                 }   

  30.                 PropertyChanged(this, new PropertyChangedEventArgs(“IsChecked”));   

  31.                 if (updateChildren) {   

  32.                     if (HasChildren()) {   

  33.                         Children.ForEach(c => c.SetCheckState(value, true, false));   

  34.                     }   

  35.                 }   

  36.                 if (updateParent && _parent != null) {   

  37.                     _parent.VerifyState();   

  38.                 }   

  39.             }   

  40.         }   

  41.         private void VerifyState() {   

  42.             bool? state = null;   

  43.             for (int i = 0; i < this.Children.Count; ++i) {   

  44.                 bool? currentState = this.Children[i].IsChecked;   

  45.                 if (i == 0) {   

  46.                     state = currentState;   

  47.                 } else if (state != currentState) {   

  48.                     state = null;   

  49.                     break;   

  50.                 }   

  51.             }   

  52.             this.SetCheckState(state, false, true);   

  53.         }   

  54.         private bool _isExpanded;   

  55.         public bool IsExpanded {   

  56.             get { return _isExpanded; }   

  57.             set {   

  58.                 if (value != _isExpanded) {   

  59.                     _isExpanded = value;   

  60.                     PropertyChanged(this, new PropertyChangedEventArgs(“IsExpanded”));   

  61.                 }   

  62.                 if (!HasChildren()) {   

  63.                     Children.Remove(_temp);   

  64.                     LoadChildren();   

  65.                 }   

  66.             }   

  67.         }   

  68.         private object _current;   

  69.         public object Current {   

  70.             get { return _current; }   

  71.             set { _current = value; }   

  72.         }   

  73.         public string DisplayText {   

  74.             get { return ((DepartmentTreeView.DB.SampleDataSet.DepartmentRow)Current)[“DName”].ToString(); }   

  75.         }   

  76.         private DepartmentViewModel _parent;   

  77.         public DepartmentViewModel Parent {   

  78.             get { return _parent; }   

  79.             set { _parent = value; }   

  80.         }   

  81.         private List _children;   

  82.         public List Children {   

  83.             get { return _children; }   

  84.             private set { _children = value; }   

  85.         }   

  86.         #endregion   

  87.         public static List Create() {   

  88.             var list = DepartmentHelper.GetSubDepartments(0);   

  89.             DepartmentViewModel root = new DepartmentViewModel(null);   

  90.             _rootItem = root;   

  91.             root.Children.Clear();   

  92.             foreach (var item in list) {   

  93.                 root.Children.Add(new DepartmentViewModel(item));   

  94.             }   

  95.             return root.Children;   

  96.         }   

  97.         private DepartmentViewModel(object currentObject) {   

  98.             Current = currentObject;   

  99.             _isChecked = false;   

  100.             Children = new List();   

  101.             Children.Add(_temp);   

  102.         }   

  103.         private void Init() {   

  104.             if (!HasChildren()) return;   

  105.             foreach (DepartmentViewModel child in Children) {   

  106.                 child.Parent = this;   

  107.                 child.Init();   

  108.             }   

  109.             PropertyChanged(this, new PropertyChangedEventArgs(“Children”));   

  110.         }   

  111.         private void LoadChildren() {   

  112.             if (Current != null) {   

  113.                 int pid = Convert.ToInt32(((DepartmentTreeView.DB.SampleDataSet.DepartmentRow)Current)[“DID”]);   

  114.                 var list = DepartmentHelper.GetSubDepartments(pid);   

  115.                 foreach (var item in list) {   

  116.                     DepartmentViewModel model = new DepartmentViewModel(item) { _isChecked = this.IsChecked };   

  117.                     if (model.IsChecked == true) {   

  118.                         _checkedItems.Add(model);   

  119.                         PropertyChanged(this, new PropertyChangedEventArgs(“CheckedItems”));   

  120.                     }   

  121.                     Children.Add(model);   

  122.                 }   

  123.                 Init();   

  124.             }   

  125.         }   

  126.         private bool HasChildren() {   

  127.             return !(Children.Count == 1 && Children[0] == _temp);   

  128.         }   

  129.         public ObservableCollection CheckedItems {   

  130.             get {   

  131.                 return _checkedItems;   

  132.             }   

  133.         }   

  134.         public event PropertyChangedEventHandler PropertyChanged;   

  135.     }   

  136. }
